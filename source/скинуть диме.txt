CMD:mute(playerid, params[])
{
    if(GetPlayerAdminLevel(playerid) < 2) return true;
    if(sscanf(params,"dis[128]", params[0],params[1],params[2])) return SendClientMessage(playerid, -1, "Используйте: /mute [id игрока] [минуты] [причина]");
    if(params[0] == INVALID_PLAYER_ID) return 1;
    if(params[1] > 200) return SendClientMessage(playerid, -1, ""COLOR_ORANGE"Максимальный бан чата - 30 минут");
    //if(GetPlayerAdminLevel(params[0]) > 0) return true;
    new
        family[MAX_PLAYER_NAME];

    /**
     * В FindFamily передаем:
     * первым параметром - причину наказания
     * вторым параметром - переменную (типа string) в которую запишется фамилия
     *
     * Функция вернет:
     * true - если игрок с указанным после "@" ID на сервере
     * false - если игрока нет на сервере
     */

    new ret_value = FindFamily(params[2], family);

    if(ret_value == true) {
        format(small_string, sizeof(small_string), ""COLOR_RED"Администратор %s заблокировал чат игроку %s на %d минут. Причина: %s "COLOR_YELLOW"by  %s", PlayerInfo[playerid][pName], PlayerInfo[params[0]][pName], params[1], params[2], family);
        SendJudgeMessageToAll(small_string, params[0]);
    } else {
        format(small_string, sizeof(small_string), ""COLOR_RED"Администратор %s заблокировал чат игроку %s на %d минут. Причина: %s", PlayerInfo[playerid][pName], PlayerInfo[params[0]][pName], params[1], params[2]);
        SendJudgeMessageToAll(small_string, params[0]);
    }

    PlayerInfo[params[0]][pMuted] = 1;
    PlayerInfo[params[0]][pMuteTime] = params[1] * 60;
    SetPlayerInt(params[0], "Muted", PlayerInfo[params[0]][pMuted]);
    SetPlayerInt(params[0], "MuteTime", PlayerInfo[params[0]][pMuteTime]);
    Admin.UpCountMute(GetPlayerAccountID(playerid));
    return true;
}
